  isValidNumber_Positive(value: any): boolean {
    if (!value) return false;
    if (isNaN(value)) return false;
    if (value <= 0) return false;
    return true;
  }

  isValidNumber_Negative(value: any): boolean {
    if (!value) return false;
    if (isNaN(value)) return false;
    if (value >= 0) return false;
    return true;
  }
